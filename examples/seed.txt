CREATE DATABASE talktosql_sample_db;
USE talktosql_sample_db;

CREATE TABLE Owners (
  owner_id INT PRIMARY KEY,
  first_name VARCHAR(50),
  last_name VARCHAR(50),
  email VARCHAR(100),
  phone VARCHAR(20)
);

CREATE TABLE Pets (
  pet_id INT PRIMARY KEY,
  owner_id INT,
  pet_name VARCHAR(50),
  species VARCHAR(50),
  breed VARCHAR(50),
  age INT,
  gender VARCHAR(10),
  FOREIGN KEY (owner_id) REFERENCES Owners(owner_id)
);
